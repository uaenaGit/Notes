 *     int val;

 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}

class Solution { 
public:
    void preOrder(TreeNode* root, vector<string> &res, string path) {
        if (!root) return; // ç©ºèŠ‚ç‚¹ç›´æ¥è¿”å›

        path.push_back(root->val + '0'); // å°†å½“å‰èŠ‚ç‚¹è®¡å…¥è·¯å¾„

        }
        if (root->left) preOrder(root->left, res, path);
        if (root -> right) preOrder(root->right, res, path);
    }
    int sumRootToLeaf(TreeNode* root) {
        vector<string> res;
        string s;
        preOrder(root, res, s);
        
        int sum = 0;
        for (auto &str : res) {
            int t = 0;
            for (int i = 0; i < str.size(); i++) {
                // 1001 = 1 << 3 + 0 << 2 + 0 << 1 + 1 << 0;
                t += (str[i] - '0') << (str.size() - i - 1);
            }
            sum += t;
        }
        return sum;
    }
};
